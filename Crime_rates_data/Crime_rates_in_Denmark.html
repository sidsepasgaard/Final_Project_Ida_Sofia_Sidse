Introduction: In this analysis, we evaluate the correlation between immigrants and registered criminal convictions in Denmark from 2000 to 2025. To evaluate whether certain groups are disproportionately represented in crime statistics without creating a mathematically distorted conclusion, we standardize the data by calculating convictions per 1,000 individuals within each specific demographic group)

##Step 1 First we load the needed visualization libraries and read the datasets.

library(pdftools)
library(tidytext)
library(tidyverse)

crime_data <- read.csv2("data/dataset_1.csv")
pop_data   <- read.csv2("data/population.csv")

options(scipen = 999)

##Step 2 For the first graph we want an overall view of conviction and population across age groups and gender. so we clean the data to get the total conviction and the total population.

crime_clean <- crime_data %>%
  group_by(year) %>%
  summarize(Danish_Crime = sum(Danish_origin, na.rm = TRUE),
    Immigrant_Crime = sum(immigrants_total, na.rm = TRUE))
pop_clean <- pop_data %>%
  group_by(year) %>%
  summarize(Danish_Pop = sum(Danish_Origin, na.rm = TRUE),
    Immigrant_Pop = sum(Immigrant, na.rm = TRUE))

##Step 3 By dividing the conviction numbers by the population and multiplying by 1000 we isolate behavior trends from group size differences. We reshape the data into long format for visualization.

analysis_data <- inner_join(crime_clean, pop_clean, by = c("year" = "year")) %>%
  mutate(Rate_Danish    = (Danish_Crime / Danish_Pop) * 1000,
    Rate_Immigrant = (Immigrant_Crime / Immigrant_Pop) * 1000)

plot_data <- analysis_data %>%
  select(year, Rate_Danish, Rate_Immigrant) %>%
  pivot_longer(cols = c(Rate_Danish, Rate_Immigrant), 
               names_to = "Origin", 
               values_to = "Crime_Rate") %>%
  mutate(Origin = if_else(Origin == "Rate_Danish", "Danish Origin", "Immigrants"))

##Step 4 Here we make the graph showing the crime rate across age group from the year 2000 to 2025, divided by danish origin and immigrants

ggplot(plot_data, aes(x = year, y = Crime_Rate, color = Origin, group = Origin)) +
  geom_line(linewidth = 1.2) +
  geom_point(size = 3) +
  scale_color_manual(values = c("Danish Origin" = "lightblue", "Immigrants" = "orange")) +
  scale_x_continuous(breaks = seq(2000, 2025, by = 2)) +
  labs(title = "Danish Crime Rate Comparison (2000-2025)", x = "Year", y = "Convictions per 1,000 People", color = "Demographic Group") +
  theme_minimal() +
  theme(plot.title = element_text(face = "bold", size = 14, hjust = 0.5),
    plot.subtitle = element_text(color = "grey", size = 10, hjust = 0.5),
    legend.position = "bottom")

##Step 5 Here we are making a graph where we can see each different age group

crime_age_clean <- crime_data %>%
  group_by(year, age) %>%
  summarize(Danish_Crime = sum(Danish_origin, na.rm = TRUE),
    Immigrant_Crime = sum(immigrants_total, na.rm = TRUE), .groups = "drop")

##Step 6 Because the total conviction is flawed by the historical demography we again calculate the rate per 1000 individual.

analysis_age_data <- inner_join(crime_age_clean, pop_data, by = c("year", "age")) %>%
  mutate(Rate_Danish = (Danish_Crime / Danish_Origin) * 1000,
    Rate_Immigrant = (Immigrant_Crime / Immigrant) * 1000)

##Step 7 The data is wide format and needs to be long format. We accomplish that with the pivot_longer() function.

plot_age_data <- analysis_age_data %>%
  select(year, age, Rate_Danish, Rate_Immigrant) %>%
  pivot_longer(cols = c(Rate_Danish, Rate_Immigrant), names_to = "Origin", values_to = "Crime_Rate") %>%
  mutate(Origin = if_else(Origin == "Rate_Danish", "Danish Origin", "Immigrants"))

##Step 8 With the facet_wrap(agegroup) function we can make a graph for each age group instead of having all the lines in one graph.

ggplot(plot_age_data, aes(x = year, y = Crime_Rate, color = Origin, group = Origin)) +
  geom_line(linewidth = 1.2) +
  geom_point(size = 2) +
  facet_wrap(~age, scales = "fixed") + 
  scale_color_manual(values = c("Danish Origin" = "lightblue", "Immigrants" = "orange")) +
  scale_x_continuous(breaks = seq(2000, 2025, by = 4)) +
  labs(
    title = "Crime Rate Comparison by age group (2000-2025)",
    subtitle = "Per 1,000 individuals within each specific age group",
    x = "Year",
    y = "Convictions per 1,000 People",
    color = "Demographic Group")+
  theme_minimal() +
  theme(plot.title = element_text(face = "bold", size = 13, hjust = 0.5),
    plot.subtitle = element_text(color = "grey", size = 9, hjust = 0.5),
    legend.position = "bottom",
    strip.text = element_text(face = "bold", size = 12, color = "darkblue"))

##Step 9 To see the difference between non-western and western immigrants, but the relative rate of change, since the population aspect isn’t a part of this graph, because there were no data on the population of non-western and western immigrants

crime_geo_clean <- crime_data %>%
  group_by(year) %>%
  summarize(Danish_Crime = sum(Danish_origin, na.rm = TRUE),
    Western_Crime = sum(get("X.immigrants_western_countries"), na.rm = TRUE),
    Non_Western_Crime = sum(Immigrant_nonwestern_countries, na.rm = TRUE))

##Step 10 Here we make the graph

plot_geo <- crime_geo_clean %>%
  pivot_longer(cols = c(Danish_Crime, Western_Crime, Non_Western_Crime), names_to = "Category", values_to = "Total_Convictions")

ggplot(plot_geo, aes(x = year, y = Total_Convictions, color = Category, group = Category)) +
  geom_line(linewidth = 1.2) +
  scale_y_log10() + 
  labs(title = "Convictions by Category (2000-2025)",
    x = "Year", y = "Total Annual Convictions (Log Scale)") +
  theme_minimal()

##Step 11 Now we wanna expand our data across age, year and gender.

crime_gender_clean <- crime_data %>%
  group_by(year, age, gender) %>%
  summarize(Danish_Crime    = sum(Danish_origin, na.rm = TRUE),
    Immigrant_Crime = sum(immigrants_total, na.rm = TRUE),
    .groups = "drop")

##Step 12 Again we change the format and make it convictions per 1000 individuals.

analysis_gender_data <- inner_join(crime_gender_clean, pop_data, by = c("year", "age")) %>%
  mutate(Rate_Danish    = (Danish_Crime / Danish_Origin) * 1000,
    Rate_Immigrant = (Immigrant_Crime / Immigrant) * 1000) %>%
  pivot_longer(cols = c(Rate_Danish, Rate_Immigrant), names_to = "Origin", values_to = "Crime_Rate") %>%
  mutate(Origin = if_else(Origin == "Rate_Danish", "Danish Origin", "Immigrants"))

##Step 13 Creating a graph with facet_grid(gender ~ age) makes it a 2-row by 3-column visualization.

ggplot(analysis_gender_data, aes(x = year, y = Crime_Rate, color = Origin, group = Origin)) +
  geom_line(linewidth = 1) +
  facet_grid(gender ~ age, scales = "free_y") +
  scale_color_manual(values = c("Danish Origin" = "lightblue", "Immigrants" = "orange")) +
  labs(title = "Crime Rates by Age and Gender (2000-2025)", 
  x = "Year", y = "Convictions per 1,000 People") +
  theme_minimal() +
  theme(legend.position = "bottom")
